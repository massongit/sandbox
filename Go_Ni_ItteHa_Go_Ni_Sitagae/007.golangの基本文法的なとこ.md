```go

// パッケージ宣言が必要
package main

// import文が必要
import "fmt"


// 関数はトップレベルに宣言できる
// funcが関数のキーワード
func main() {

    // 変数宣言はvar。
    // var {変数名} {型}  = {値}
    var a string = "string"
    fmt.Println(a)

    // 変数2つの代入も可能
    var b, c int = 1, 2
    fmt.Println(b, c)


    // boolean値
    var d = true
    fmt.Println(d)

　　// int値。初期化しない場合は0が入る
    var e int
    fmt.Println(e)

    // じゃあstringはどうなるかというと空文字のようだ
    var s string
    fmt.Println(s)

    // これは型推論かな
    f:= "short"
    fmt.Println(f)
}

```


出力結果
```
string
1 2
true
0

short
```
